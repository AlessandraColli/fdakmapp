152 class AffineFunction : public WarpingFunction
153 {
154 
155 public:
156  virtual uword n_pars()
157  {
158  return 2;
159  }
160 
161  virtual mat apply_warping(const mat& x, const mat& par )
162  {
163  mat out(x.n_rows,x.n_cols);
164  for(size_t obs=0; obs < x.n_rows; obs++)
165  {
166  out.row(obs) = par(0,obs)*x.row(obs) + par(1,obs);
167  }
168  return out;
169  }
170  virtual void set_bounds(const rowvec& war_opt, const mat& x)
171  {
172  double min_temp= min( util::uppers(x)- util::lowers(x) );
173  double dl= war_opt(0), sl = war_opt(1);
174  upperBound = { 1 + dl, sl * min_temp};
175  lowerBound = { 1 - dl, -sl * min_temp};
176  }
177  virtual mat final_warping(const cube& parameters_vec, const urowvec& labels, const urowvec& ict);
178 
179  virtual void normalize(mat& par,const urowvec& ict,const urowvec& labels);
180 
181  virtual double warp(const warping_set& w_set, const colvec& arg) const
182  {
183  return diss->compute(arg(0)*w_set.xf+arg(1),w_set.xg,w_set.yf,w_set.yg);
184  }
185 
186 };
